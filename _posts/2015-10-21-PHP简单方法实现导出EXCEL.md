description: 生成excel 当然使用的是 phpExcel这个类库了,可是它太麻烦了,对于只要简单生成来说有点不值得。
---

```language-php
<?php
namespace Home\Controller;
use Think\Controller;
class IndexController extends Controller {
    public function index(){
    	header("Content-type: text/html; charset=utf-8");
    	$Store = M('store');
    	$Goods = M('goods');
    	$list = $Store->field('store_id,store_name,store_state')->select();
    	foreach ($list as $key => $val) {
    		$list[$key]['count'] = $Goods->where(array('store_id'=>$val['store_id']))->count();
    	}
    	$filename =  iconv("UTF-8", "GBK", "陶博会店铺商品资料");
    	header("Content-type:application/vnd.ms-excel");
        header("Content-Disposition:attachment;filename=$filename.xls");
		echo  iconv("UTF-8", "GBK","店铺名")."\t"; 
		echo  iconv("UTF-8", "GBK","商品数量")."\t"; 
		echo  "\n"; 
		foreach ($list as $key => $val) {
			echo iconv("UTF-8", "GBK", $val['store_name']).chr(30)."\t";
		    echo $val['count'].chr(30)."\t";
		    echo "\n"; 
		}
    }
}
```
